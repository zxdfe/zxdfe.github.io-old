class Solution {
public:
    int numIslands(vector<vector<char>>& grid) {
        this->m = grid.size();
        this->n = grid[0].size();
        visited = vector<vector<bool>>(m, vector<bool>(n, false));
        int ans = 0;
        for (int i = 0; i < m; i++) 
            for (int j = 0; j < n; j++)
                if (grid[i][j] == '1' && !visited[i][j]) {
                    bfs(grid, i, j);
                    ans++;
                }
        return ans;
    }

private:
    void bfs(vector<vector<char>>& grid, int sx, int sy) {
        queue<pair<int, int>> q;
        q.push({sx, sy});
        visited[sx][sy] = true;
        while (!q.empty()) {
            // 1. 取队头
            int x = q.front().first;
            int y = q.front().second;
            q.pop();
            // 2. 扩展队头
            for (int i = 0; i < 4; i++) {
                int nx = x + dx[i];
                int ny = y + dy[i];
                if (nx < 0 || ny < 0 || nx >= m ||  ny >= n) continue;
                // if (grid[nx][ny] != '1') continue;
                // if (visited[nx][ny]) continue;
                // q.push({nx, ny});
                // visited[nx][ny] = true;
                if (grid[nx][ny] == '1' && !visited[nx][ny]) {
                    q.push({nx, ny});
                    visited[nx][ny] = true;
                }
            }
        }
    }
    int m, n;
    vector<vector<bool>> visited;
    // 方向数组
    const int dx[4] = {-1, 0, 0, 1};
    const int dy[4] = {0, -1, 1, 0};
};

class Solution {
public:
    int minMutation(string start, string end, vector<string>& bank) {
        // 广搜求层数
        depth[start] = 0;
        for (string& seq : bank) hashBank.insert(seq);
        // end not in hashBank, return -1
        if (hashBank.find(end) == hashBank.end()) return -1;
        queue<string> q;
        q.push(start);
        const char gene[4] = {'A', 'C', 'G', 'T'};
        while (!q.empty()) {
            string s = q.front();
            q.pop();
            for (int i = 0; i < 8; i++)
                for (int j = 0; j < 4; j++) 
                    if (s[i] != gene[j]) {
                        string ns = s; // 拷贝
                        ns[i] = gene[j];
                        // ns not in bank, continue
                        if (hashBank.find(ns) == hashBank.end()) continue;
                        // ns in depth
                        // 每个点只需要访问一次, 第一次就是最少层数
                        if (depth.find(ns) != depth.end()) continue;
                        depth[ns] = depth[s] + 1;
                        q.push(ns);
                        if (ns == end) {
                            return depth[ns];
                        }
                    }
        }
        return -1;
    }

private:
    unordered_set<string> hashBank; // 无序集合求包含比数组快
    unordered_map<string, int> depth;
};

class Solution {
public:
    int longestIncreasingPath(vector<vector<int>>& matrix) {
        m = matrix.size();
        n = matrix[0].size();
        to = vector<vector<int>> (m * n);
        deg = vector<int>(m *n, 0);
        dist = vector<int>(m *n, 0);
        const int dx[4] = {-1, 0, 0, 1};
        const int dy[4] = {0, -1, 1, 0};
        for (int i = 0; i < m; i++) 
            for (int j = 0; j < n; j++) {
                for (int k = 0; k < 4; k++) {
                    int ni = i + dx[k];
                    int nj = j + dy[k];
                    if (valid(ni, nj) && matrix[ni][nj] > matrix[i][j]) {
                        addEdge(num(i, j), num(ni, nj));
                    }
                }
            }
        queue<int> q;
        for (int i = 0; i < m * n; i++)
            if (deg[i] == 0) {
                q.push(i);
                dist[i] = 1;
            }
        while (!q.empty()) {
            int x = q.front();
            q.pop();
            for (int y : to[x]) {
                deg[y]--;
                dist[y] = max(dist[y], dist[x] + 1);
                if (deg[y] == 0) q.push(y);
            }
        }
        int ans = 0;
        for (int i = 0; i < m *n; i++) ans = max(ans, dist[i]);
        return ans;
    }
private:
    int m, n;
    vector<vector<int>> to;
    vector<int> deg;
    vector<int> dist;

    void addEdge(int u, int v) {
        deg[v]++;
        to[u].push_back(v);
    }
    // 把二维行列格子坐标 变为一维的一个点的编号
    int num(int i, int j) {
        return i * n + j;
    }

    bool valid(int i, int j) {
        return i >= 0 && i < m && j >=0 && j < n;
    }
};

class Solution {
    public int longestIncreasingPath(int[][] matrix) {
        this.matrix = matrix;
        m = matrix.length;
        n = matrix[0].length;
        dist = new int[m][n]; // default value: 0 表示没算过
        dx = new int[]{-1, 0, 0, 1};
        dy = new int[]{0, -1, 1, 0};
        int ans = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                ans = Math.max(ans, dfs(i, j));
            }
        }
        return ans;
    }

    int dfs(int x, int y) {
        if (dist[x][y] != 0) return dist[x][y];
        dist[x][y] = 1;
        for (int k = 0; k < 4; k++) {
            int nx = x + dx[k];
            int ny = y + dy[k];
            if (valid(nx, ny) && matrix[nx][ny] > matrix[x][y]) {
               dist[x][y] = Math.max(dist[x][y], dfs(nx, ny) + 1); 
            }
        }
        return dist[x][y];
    }

    boolean valid(int i, int j) {
        return i >= 0 && i < m && j >=0 && j < n;
    }

    int[][] matrix;
    int m, n;
    int[] dx;
    int[] dy;
    int[][] dist;
}
